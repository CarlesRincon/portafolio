document.addEventListener("DOMContentLoaded", function () { const container = document.querySelector(".elementor-container"); const loadingPage = document.getElementById("loadingPage"); const cv = document.getElementById("cv"); const skills = document.querySelector(".cont-skills"); const portfolio = document.getElementById("portfolio"); const sections = [loadingPage, cv, skills, portfolio]; let currentIndex = 0; let isScrolling = false; function scrollToSection(index) { if (index = sections.length) return; isScrolling = true; currentIndex = index; let offset = sections[index].offsetLeft; // Ajuste especial para cont-skills if (index === 2) { offset -= window.innerWidth * 0.4; // Ajuste al ir a skills } else if (index === 1) { offset = sections[index].offsetLeft; // Corrige la posición al volver a CV } container.scrollTo({ left: offset, behavior: "smooth" }); setTimeout(() => { isScrolling = false; }, 800); } document.addEventListener("wheel", function (event) { if (isScrolling) return; let direction = event.deltaY > 0 ? 1 : -1; let newIndex = Math.min(Math.max(currentIndex + direction, 0), sections.length - 1); if (newIndex !== currentIndex) { scrollToSection(newIndex); } event.preventDefault(); }, { passive: false }); // Asegura que al recargar la página esté en loadingPage window.addEventListener("load", function () { container.scrollLeft = sections[currentIndex].offsetLeft; }); }); //menu document.addEventListener("DOMContentLoaded", function () { let menu = document.getElementById("menu-lateral"); document.addEventListener("mousemove", function (event) { if (event.clientX > window.innerWidth - 100) { menu.style.right = "0"; // Muestra el menú } else { menu.style.right = "-150px"; // Lo oculta } }); }); //SKILLS document.addEventListener("DOMContentLoaded", function () { const skillsSection = document.querySelector(".skills-container"); function resetSkillsAnimation() { document.querySelectorAll(".skill").forEach(skill => { let level = skill.getAttribute("data-level") + "%"; let liquid = skill.querySelector(".skill-liquid"); // Reinicia la animación reseteando el nivel liquid.style.height = "1%"; setTimeout(() => { liquid.style.height = level; }, 800); }); } // Configuración de Intersection Observer const observer = new IntersectionObserver(entries => { entries.forEach(entry => { if (entry.isIntersecting) { resetSkillsAnimation(); } }); }, { threshold: 0.5 }); // Se activa cuando al menos el 50% del contenedor sea visible observer.observe(skillsSection); });
